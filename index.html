GOOGLE SHEETS

Apps Script 웹앱 URL을 입력하면 구글 시트에 데이터가 저장됩니다.

설정 방법 (최초 1회)
1. 📊 공개보고서 구글 시트 열기
2. 확장 프로그램 → Apps Script
3. 아래 코드 복사 → 저장 → 배포 → 새 배포
4. 웹앱 · 모든 사용자 → 배포 → URL 입력
📋 Apps Script 코드 보기 
// ── 필수 설정값 ──────────────────────────────
const SHEET_ID   = 'YOUR_SPREADSHEET_ID';  // ← 구글 시트 ID 입력
const SHEET_NAME = '전도보고서';

// 알림 수신 이메일 목록
const ADMIN_EMAILS = ['admin@example.com'];
const SEND_EMAIL = true;

// ── doPost : 보고서 저장 + 이메일 알림 ──────
function doPost(e){
  try{
    const d=JSON.parse(e.postData.contents);
    if(d.action==='save_jo'){
      PropertiesService.getScriptProperties().setProperty('JO_DATA',JSON.stringify(d.joData));
      return ok({});
    }
    const ss=SpreadsheetApp.openById(SHEET_ID);
    const s=ss.getSheetByName(SHEET_NAME)||ss.insertSheet(SHEET_NAME);
    if(s.getLastRow()===0) writeHeader(s);
    const savedAt=Utilities.formatDate(new Date(),'Asia/Seoul','yyyy-MM-dd HH:mm:ss');
    s.appendRow([
      savedAt,d.week,d.date,d.doc_number,d.evang_type,d.evang_count,
      d.jo,d.leader,d.trainer,d.trainee1,
      d.contact,d.gospel,d.target,d.decision,d.reject,d.confirm,d.believer,
      d.location,d.name,d.age,d.marriage,d.gender,
      d.meet_place,d.gospel_place,d.target_type,d.relation_member,
      d.religion,d.job,d.heaven,d.reason,d.yeopjeop,d.diag_result,
      d.testimony6,d.testimony7,
      d.leader_part,d.leader_action,d.trainer_part,d.trainer_action,
      d.trainee_part,d.trainee_action,
      d.result,d.immed_nurture,d.followup,d.followup_reason,
      d.q10,d.q11,d.q12,d.q13
    ]);
    if(SEND_EMAIL) sendEmailAlert(d,savedAt);
    return ok({});
  }catch(err){return error(err.toString());}
}

// ── doGet : 기록 조회 ────────────────────────
function doGet(e){
  try{
    const action=e&&e.parameter&&e.parameter.action;
    if(action==='get_jo'){
      const raw=PropertiesService.getScriptProperties().getProperty('JO_DATA');
      return ok({joData:raw?JSON.parse(raw):[]});
    }
    const s=SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME);
    if(!s||s.getLastRow()<2) return ok({records:[]});
    const rows=s.getDataRange().getValues();
    const h=rows[0];
    const rec=rows.slice(1).map((r,i)=>{const o={rowIndex:i+2};h.forEach((k,j)=>o[k]=r[j]);return o;}).reverse();
    return ok({records:rec});
  }catch(err){return error(err.toString());}
}

// ── 이메일 알림 ──────────────────────────────
function sendEmailAlert(d,savedAt){
  const subject=`[전도보고서] ${d.jo}조 ${d.leader} 팀 (${savedAt.slice(0,10)})`;
  const body=
`━━━━━━━━━━━━━━━━━━━━━━
현장전도 보고서가 접수되었습니다
━━━━━━━━━━━━━━━━━━━━━━
제출일시 : ${savedAt}
전도일   : ${d.date||'-'}
조       : ${d.jo||'-'}조
훈련자   : ${d.leader||'-'}
훈련생   : ${d.trainee1||'-'}

[통계]
접촉: ${d.contact||0}회 | 복음제시: ${d.gospel||0}회
결신: ${d.decision||0}명 | 보류/거절: ${d.reject||0}명

[진단]
천국확신: ${d.heaven||'-'} | 이유: ${d.reason||'-'}
진단결과: ${d.diag_result||'-'}
즉석양육: ${d.immed_nurture||'-'}
후속양육: ${d.followup||'-'}

[특별경험] ${d.q10||'(없음)'}
[문제점]   ${d.q11||'(없음)'}

모든 영광을 하나님께 올려드립니다!`;
  ADMIN_EMAILS.forEach(email=>{
    try{GmailApp.sendEmail(email,subject,body);}
    catch(err){Logger.log('이메일 오류[%s]:%s',email,err);}
  });
}

// ── 헤더 / 유틸 ─────────────────────────────
function writeHeader(s){
  s.appendRow(['저장일시','주차','전도일','문서번호','전도유형','횟수',
    '조','훈련자(조장)','훈련자','훈련생',
    '접촉횟수','복음제시','대상자','결신','보류/거절','확신','이미신자',
    '거주지역','성씨','나이','결혼여부','성별','만난장소','복음제시장소',
    '대상자구분','관계조원','종교배경','직업',
    '천국확신','이유대답','영접여부','진단결과','간증6','간증7',
    '훈련자(조장)역할','훈련자(조장)행동','훈련자역할','훈련자행동','훈련생역할','훈련생행동',
    '결과','즉석양육','후속양육','하지못한사유',
    '느낀점(10)','문제점(11)','기도제목(12)','기타(13)']);
  const hr=s.getRange(1,1,1,s.getLastColumn());
  hr.setBackground('#1a3a9e').setFontColor('#fff').setFontWeight('bold').setHorizontalAlignment('center');
  s.setFrozenRows(1);
}
function ok(d){return ContentService.createTextOutput(JSON.stringify(Object.assign({status:'ok'},d))).setMimeType(ContentService.MimeType.JSON);}
function error(m){return ContentService.createTextOutput(JSON.stringify({status:'error',message:m})).setMimeType(ContentService.MimeType.JSON);}
